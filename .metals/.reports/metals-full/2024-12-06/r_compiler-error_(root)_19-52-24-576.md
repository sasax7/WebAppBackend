import javax.inject.{Inject, Singleton}
import scala.concurrent.{ExecutionContext, Future}
import models.StrategyDetails

import play.api.db.slick.DatabaseConfigProvider
import slick.jdbc.JdbcProfile

@Singleton
class StrategyRepository @Inject() (
    dbConfigProvider: DatabaseConfigProvider
)(implicit ec: ExecutionContext) {
  private val dbConfig = dbConfigProvider.get[JdbcProfile]
  private val db = dbConfig.db

  def getStrategiesWithDetailsByFirebaseUid(firebaseUid: String): Future[Seq[StrategyDetails]] = {
    val query =
      sql"""
        SELECT 
            s.id AS strategy_id,
            s.name AS strategy_name,
            ARRAY_AGG(DISTINCT i.name) AS indicator_names,
            ARRAY_AGG(DISTINCT sl.name) AS stop_loss_names,
            ARRAY_AGG(DISTINCT tp.name) AS take_profit_names,
            ARRAY_AGG(DISTINCT pp.name) AS price_point_names
        FROM strategies s
        JOIN userstrategies us ON s.id = us.strategy_id
        JOIN users u ON us.user_id = u.id
        LEFT JOIN strategyindicator si ON s.id = si.strategy_id
        LEFT JOIN indicatorname i ON si.indicator_id = i.id
        LEFT JOIN strategystoplosses ssl ON s.id = ssl.strategy_id
        LEFT JOIN stoplossesname sl ON ssl.stoploss_id = sl.id
        LEFT JOIN strategytakeprofit stp ON s.id = stp.strategy_id
        LEFT JOIN takeprofitname tp ON stp.takeprofit_id = tp.id
        LEFT JOIN strategypricepoint spp ON s.id = spp.strategy_id
        LEFT JOIN pricepointname pp ON spp.pricepoint_id = pp.id
        WHERE u.firebase_uid = $firebaseUid
        GROUP BY s.id, s.name
      """
        .as[(Long, String, Seq[String], Seq[String], Seq[String], Seq[String])]

    db.run(query).map(_.map {
      case (strategyId, strategyName, indicatorNames, stopLossNames, takeProfitNames, pricePointNames) =>
        StrategyDetails(strategyId, strategyName, indicatorNames, stopLossNames, takeProfitNames, pricePointNames)
    })
  }
}
